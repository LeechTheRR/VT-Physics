PROJECT(VPLibs-logger)

FIND_PACKAGE(spdlog CONFIG REQUIRED)

FILE(GLOB_RECURSE LIB_LOGGER_HEADER_FILES
        ${VP_LIB_ROOT}/Include/Logger/*.hpp
)
FILE(GLOB_RECURSE LIB_LOGGER_SRC_FILES
        ${VP_LIB_ROOT}/Source/Logger/*.cpp
)

ADD_LIBRARY(vplogger STATIC ${LIB_LOGGER_HEADER_FILES} ${LIB_LOGGER_SRC_FILES})
TARGET_LINK_LIBRARIES(vplogger PUBLIC spdlog::spdlog)
